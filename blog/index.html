/**
* @param {number[]} nums
* @param {number} target
* @return {number[]}
*/
var twoSum = function (nums, target) {
let result = {};
for (let index = 0; index < nums.length; index++) {
let diff = target - nums[index];
if(Object.keys(result).includes(String(diff))) {
return [result[diff], index]
}
result[nums[index]] = index
}
};

* @param {string} s
* @param {number} numRows
* @return {string}
*/
var convert = function(s, numRows) {
let isdown = false;
let temp = [];
let step = 0;
if(numRows === 1) return s;
for(let i = 0; i< s.length; i++) temp[i] = '';
for(let index = 0; index < s.length; index++) {
temp[step] += s[index];
if(step === 0 || step === numRows -1) {
isdown = !isdown
}
step += isdown ? 1 : -1;
}
return temp.join('')
};

/**
* @param {number[]} height
* @return {number}
*/
var maxArea = function (height) {
let maxAreas = 0;
let left = 0;
let right = height.length - 1;
while (right > left) {
const temp = Math.min(height[right], height[left]) * (right - left);
if (height[left] > height[right]) {
right--;
} else {
left++;
}
maxAreas = Math.max(temp, maxAreas);
}
return maxAreas;
};

/**
* @param {string} s
* @return {boolean}
*/
var isValid = function (s) {
let match = {
")": "(",
"}": "{",
"]": "[",
};
let stack = [];
if (s.length % 2 === 1) {
return false;
}
for (let i = 0; i < s.length; i++) {
if (!match[s[i]]) {
stack.push(s[i]);
} else {
if (stack[stack.length - 1] === match[s[i]]) {
stack.pop();
} else {
return false;
}
}
}
return stack.length === 0;
};

var mergeTwoLists = function(l1, l2) {
const element = new ListNode(-1);
let prev = element
while(l1!==null && l2!==null) {
if(l1.val <=l2.val) {
prev.next = l1
l1 = l1.next;
} else {
prev.next = l2
l2 = l2.next;

prev = prev.next

prev.next = l1 === null ? l2 : l1;

* @param {TreeNode} root
* @return {TreeNode}
*/
var invertTree = function (root) {
if (root === null) {
return root;
}
[root.left, root.right] = [invertTree(root.right), invertTree(root.left)];
return root;

while(root){

root = root.left
} else {
return root
}
}

// @lc code=start
/**
* Definition for a binary tree node.
* function TreeNode(val) {
* this.val = val;
* this.left = this.right = null;
* }
*/
/**
* @param {TreeNode} root
* @param {TreeNode} p
* @param {TreeNode} q
* @return {TreeNode}
*/
var lowestCommonAncestor = function(root, p, q) {
if(root === null || root=== q || root === p){
return root
}
const left = lowestCommonAncestor(root.left, p, q);
const right = lowestCommonAncestor(root.right, p, q);
return !left? right : !right ? left: root
};
